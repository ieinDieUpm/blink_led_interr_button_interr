93 void port_led_timer_delay_ms(uint32_t delay_ms)
94 {
95  // Disable the timer
96  TIM2->CR1 &= ~TIM_CR1_CEN;
97 
98  TIM2->CNT = 0;
99 
100  // Compute ARR and PSC to match the duration in milliseconds. Check if the duration is too long and adapt prescaler and ARR
101  double ms = (double)delay_ms;
102  double scc = (double)SystemCoreClock;
103  double psc = round((((scc / 1000.0) * ms) / (65535.0 + 1.0)) - 1.0);
104  double arr = round((((scc / 1000.0) * ms) / (psc + 1.0)) - 1.0);
105 
106  // Adjust psc and arr if necessary
107  if (arr > 0xFFFF)
108  {
109  psc += 1.0;
110  arr = round(((scc / 1000.0) * ms) / (psc + 1.0) - 1.0);
111  }
112 
113  // Load the values
114  TIM2->ARR = (uint32_t)(round(arr));
115  TIM2->PSC = (uint32_t)(round(psc));
