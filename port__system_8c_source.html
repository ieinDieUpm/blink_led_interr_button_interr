50 static void system_clock_config(void)
51 {
53  /* Power controller (PWR) */
54  /* Control the main internal voltage regulator output voltage to achieve a trade-off between performance and power consumption when the device does not operate at the maximum frequency */
55  PWR->CR &= ~PWR_CR_VOS; // Clean and set value
56  PWR->CR |= (PWR_CR_VOS & (POWER_REGULATOR_VOLTAGE_SCALE3 << PWR_CR_VOS_Pos));
57 
58  /* Initializes the RCC Oscillators. */
59  /* Adjusts the Internal High Speed oscillator (HSI) calibration value.*/
60  RCC->CR &= ~RCC_CR_HSITRIM; // Clean and set value
61  RCC->CR |= (RCC_CR_HSITRIM & (RCC_HSI_CALIBRATION_DEFAULT << RCC_CR_HSITRIM_Pos));
62 
63  /* RCC Clock Config */
64  /* Initializes the CPU, AHB and APB buses clocks */
65  /* To correctly read data from FLASH memory, the number of wait states (LATENCY)
66  must be correctly programmed according to the frequency of the CPU clock
67  (HCLK) and the supply voltage of the device. */
68 
69  /* Increasing the number of wait states because of higher CPU frequency */
70  FLASH->ACR = FLASH_ACR_LATENCY_2WS; /* Program the new number of wait states to the LATENCY bits in the FLASH_ACR register */
71 
72  /* Change in clock source is performed in 16 clock cycles after writing to CFGR */
73  RCC->CFGR &= ~RCC_CFGR_SW; // Clean and set value
74  RCC->CFGR |= (RCC_CFGR_SW & (RCC_CFGR_SW_HSI << RCC_CFGR_SW_Pos));
75 
76  /* Update the SystemCoreClock global variable */
77  SystemCoreClock = HSI_VALUE >> AHBPrescTable[(RCC->CFGR & RCC_CFGR_HPRE) >> RCC_CFGR_HPRE_Pos];
78 
79  /* Configure the source of time base considering new system clocks settings */
80  SysTick_Config(SystemCoreClock / (1000U / TICK_FREQ_1KHZ)); /* Set Systick to 1 ms */
81 }
