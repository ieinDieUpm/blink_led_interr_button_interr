100 void port_led_timer_delay_ms(uint32_t delay_ms)
101 {
102 
103  // Stop timer with interrupt enabled
104  HAL_TIM_Base_Stop_IT(&handler_timer_led);
105 
106  // Reset counter value
107  __HAL_TIM_SET_COUNTER(&handler_timer_led, 0);
108 
109  // Compute ARR and PSC to match the duration in milliseconds. Check if the duration is too long and adapt prescaler and ARR
110  double ms = (double)delay_ms;
111  double scc = (double)SystemCoreClock;
112  double psc = round((((scc / 1000.0) * ms) / (65535.0 + 1.0)) - 1.0);
113  double arr = round((((scc / 1000.0) * ms) / (psc + 1.0)) - 1.0);
114 
115  // Adjust psc and arr if necessary
116  while (arr > 0xFFFF)
117  {
118  psc += 1.0;
119  arr = round(((scc / 1000.0) * ms) / (psc + 1.0) - 1.0);
120  }
121 
122  // Update TIM2 initialization structure with computed PSC and ARR values
123  handler_timer_led.Init.Prescaler = (uint32_t)(psc);
124  handler_timer_led.Init.Period = (uint32_t)(arr);
125 
126  // Initialize the timer with the new PSC and ARR values
127  HAL_TIM_Base_Init(&handler_timer_led);
128 
129  // The PSC and ARR values are currently in the preload registers. To load them into the active registers we
130  // need an update event. We can do this manually as follows (or we could wait for the timer to expire).
131  // Generate an update event to load the new PSC and ARR values immediately
132  HAL_TIM_GenerateEvent(&handler_timer_led, TIM_EVENTSOURCE_UPDATE); // 6) Update generation: Re-inicializa el contador y actualiza los registros. IMPORTANTE que esté lo último
